import java.util.ArrayList;
import java.util.List;
import java.util.Random;
import java.util.Scanner;

public class Lotto {
    public static void main(String[] args) {
        Random rand = new Random();
        int number;
        int counter=0;
        long losowan = 0;
        long start=System.currentTimeMillis();

        List<Integer> a = new ArrayList<Integer>();
        List<Integer> b = new ArrayList<Integer>();
        List<Integer> c = new ArrayList<Integer>();

        Scanner input = new Scanner(System.in);

        for(int j=1;j<=6;j++) {
            System.out.print("Wprowadz liczbe nr."+j);
            int my = input.nextInt();
            b.add(my);
        }

    while(counter!=6) {
        losowan++;
        counter=0;

        for (int i = 1; i <= 6; i++) {
            number = rand.nextInt(49);
            if (!a.contains(number)) {
                a.add(number);
                if (b.contains(number)) {
                    counter++;
                    c.add(number);
                }
            }
        }

        if (counter == 6) {
            System.out.println("Trafiles " + counter + " razy");
            System.out.println("Trafione liczby to" + c);

        }

        c.clear();
        a.clear();

    }
    System.out.println("Potrzebnych bylo "+losowan+" losowan");
    long koniec=System.currentTimeMillis();
    long czas = ((koniec-start)/1000);

    System.out.println("Potrzebnych bylo "+czas+" sekund");
    }
}
